파이썬에서 힙소트를 구현하는 코드 입니다.

```python
import heapq

def heapsort(iterable):
    h = []
    result = []

    for value in iterable:
        heapq.heappush(h, value)

    for i in range(len(h)):
        result.append(heapq.heappop(h))
    return result

n = 8
arr = [7, 5, 2, 9, 8, 3, 4, 1]

res = heapsort(arr)

for i in range(n):
    print(res[i])
```
